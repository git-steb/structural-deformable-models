351 bool MappingSensor::performUpdate() {
352  if(!isModified()) return false;
353  if(source->getDim1Size()>0) {
354  bool doanalysis = true;
355  switch(m_MappingID) {
356  case MS_MCLAMPU:
357  case MS_GAUSSNORM:
358  case MS_MGAUSSNORM:
359  case MS_BIAS:
360  if(m_Param.size() < 4) m_Param.resize(4);
361  m_Param[1] = m_Param[2] = m_Param[3] = 0.f;
362  break;
363  default:
364  doanalysis = false;
365  }
366  if(doanalysis) {
367  int i,j;
368  for(j=0; j<source->getDim2Size(); j++) {
369  for(i=0; i<source->getDim1Size(); i++)
370  {
371  float v = source->getValue(i,j);
372  switch(m_MappingID) {
373  case MS_BIAS:
374  case MS_GAUSSNORM:
375  m_Param[1] += v;
376  m_Param[2] += v*v;
377  m_Param[3]++;
378  break;
379  case MS_MCLAMPU:
380  case MS_MGAUSSNORM:
381  if(v>0) {
382  m_Param[1] += v;
383  m_Param[2] += v*v;
384  m_Param[3]++;
385  }
386  break;
387  }
388  }
389  }
390  switch(m_MappingID) {
391  case MS_MCLAMPU:
392  m_Param[1] /= m_Param[3];
393  m_Param[2] /= m_Param[3];
394  m_Param[2] = sqrt(m_Param[2]-m_Param[1]*m_Param[1]);
395  m_Param[1] = m_Param[1]+m_Param[2]*m_Param[0];
396  break;
397  case MS_BIAS:
398  case MS_GAUSSNORM:
399  m_Param[1] /= m_Param[3];
400  m_Param[2] /= m_Param[3];
401  m_Param[2] = sqrt(m_Param[2]-m_Param[1]*m_Param[1]);
402  m_Param[2] = 1/(m_Param[2]*m_Param[0]);
403  break;
404  case MS_MGAUSSNORM:
405  m_Param[1] /= m_Param[3];
406  m_Param[2] /= m_Param[3];
407  m_Param[2] = sqrt(m_Param[2]-m_Param[1]*m_Param[1]);
408  m_Param[2] = 1/(m_Param[2]*m_Param[0]);
409  m_Param[1] = m_Param[1] - (m_Param[1]/m_Param[2]);
410  break;
411  }
412  }
413  }
414  return PPSensor::performUpdate();
415 }

419 Image<double>& create2DGaussian(Image<double>& flt, double stdev,
420  float clampd, float cx, float cy)
421 {
422  clampd = clampd*clampd;
423  int i,j;
424  double sum=0;
425  double* pixel = flt.getData();
426  int xsize=flt.getSizeX(), ysize=flt.getSizeY();
427  TRACE("creating filter " << xsize << " " << ysize);
428  for(j=0; j<ysize; j++)
429  for(i=0; i<xsize; i++, pixel++) {
430  float ii = (float)i-cx;
431  float jj = (float)j-cy;
432  float d2 = ii*ii+jj*jj;
433  if(d2<clampd) {
434  double value = gaussd2(d2, stdev);
435  *pixel = value;
436  sum += value;
437  }
438  }
439  flt *= (1.f/sum);
440  return flt;
441 }
