461 bool StructTable::findBestConnection(StructPath& path,
462  const Winner& winner) const
463 {
464  if(winner.m_WinnerID == Winner::WID_EMPTY) { TRACE0; return false; }
465  const string& csname = winner.m_StructName;
466  if(csname.empty()) { TRACE0; return false; }
467  if(path[csname] != Winner::WID_EMPTY) {
468  return false; // termination criterion
469  }
470  if(winner.m_WinnerID == Winner::WID_FRAME) {
471  path[csname] = winner.m_WinnerID;
472  return true; // confirm, but don't add goodness
473  }
474  NodePaths &np = m_NodePaths[csname];
475  NodePaths::const_iterator wpath = np.find(winner.m_WinnerID);
476  if(wpath != np.end()) { // we already have this path!
477  path.merge(wpath->second);
478  return true;
479  }
480  StructPath &bestpath = np[winner.m_WinnerID];
481  bestpath = path.branch();
482 
483  const MStructure& cstruct = getStructure(csname);
484  bestpath[csname] = winner.m_WinnerID;
485  bestpath.m_Goodness += cstruct.getWeight()
486  *winner.m_Model->getQualityOfFit();
487  //* cstruct.m_Searcher.relativeQOF(winner.m_Model->getQualityOfFit());
488  bestpath.m_Maxness += cstruct.getWeight();
489 
490  for(Winner::Ratings::const_iterator wr = winner.m_Ratings.begin();
491  wr != winner.m_Ratings.end(); wr++)
492  { // for each rating structure
493  map<string,MStructure>::const_iterator sti =
494  m_Structs.find(wr->first);
495  if(sti != m_Structs.end()) {
496  const MStructure& refstruct = sti->second;
497  const SubStructure& subs =
498  refstruct.m_SubStructures.find(csname)->second;
499  StructPath spath = bestpath;
500  for(map<dword,float>::const_iterator wi = wr->second.begin();
501  wi != wr->second.end(); wi++)
502  {
503  const Winner* nextWinner = refstruct.getWinner(wi->first);
504  StructPath cpath = spath;
505  if(nextWinner && wi->second > subs.m_RateTH) {
506  if(findBestConnection(cpath, *nextWinner)) {
507  cpath.m_Goodness += subs.m_RateWeight*wi->second;
508  cpath.m_Maxness += subs.m_RateWeight;
509  }
510  }
511  if(cpath.m_Goodness > bestpath.m_Goodness)
512  bestpath = cpath;
513  }
514  }
515  }
516  path.merge(bestpath);
517  return true;
518 }
