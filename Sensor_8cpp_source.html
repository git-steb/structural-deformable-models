141 void Sensor::calcMinMax() {
142  int i,j;
143  minval=numeric_limits<float>::max();
144  maxval=numeric_limits<float>::min();
145  double avg=0, avg2=0;
146  double size = double(source->getDim1Size()*source->getDim2Size());
147  if(size > 0.) {
148  for(j=0; j<source->getDim2Size(); j++) {
149  for(i=0; i<source->getDim1Size(); i++) {
150  float val = getValue(i,j);
151  if(val>maxval) maxval = val;
152  if(val<minval) minval = val;
153  avg+=double(val);
154  avg2+=double(val*val);
155  }
156  }
157  avg /= size;
158  avg2 /= size;
159  } else {
160  minval = maxval = 0;
161  }
162  mean = float(avg);
163  stdev = float(sqrt(avg2-avg*avg))*1.5;
164 }
